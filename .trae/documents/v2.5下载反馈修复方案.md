# v2.5 APK下载修复方案（借鉴azhon/AppUpdate设计）

## 一、根因与设计借鉴

### 根因

```
下载成功 → 存入私有目录（getExternalFilesDir）
→ saveToPublicDownloads() 静默失败（Android 6-9无权限，10+ MediaStore可能不支持）
→ installApk() 被 PDA 系统拦截（无异常抛出）
→ Dialog 消失（collector 收到 Completed 后关闭弹窗）
→ 用户找不到 APK，无任何提示 → 以为下载失败
```

### 从 azhon/AppUpdate 借鉴的设计

| azhon/AppUpdate 特性 | 借鉴方案 |
|:---------------------|:---------|
| 通知栏下载进度 | ✅ 添加 Notification 进度条（独立于弹窗） |
| 下载完成通知（点击安装） | ✅ 下载完成发送通知，PendingIntent 跳转安装器 |
| 后台下载（离开页面继续） | ✅ 通知栏进度不依赖弹窗 |
| 无需存储权限 | ✅ 当前 `getExternalFilesDir` 已经不需要 |
| Android 13 通知适配 | ✅ 声明 POST_NOTIFICATIONS + 运行时请求 |
| Toast 提示 | ✅ 下载完成/失败显示 Toast |

### 修改 2：新建 `ApkInstallReceiver.kt`

```kotlin
package com.kuaimai.pda.update

import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent
import android.net.Uri
import androidx.core.content.FileProvider
import java.io.File

/** 从通知栏点击"安装"时触发 */
class ApkInstallReceiver : BroadcastReceiver() {
    override fun onReceive(context: Context, intent: Intent) {
        val filePath = intent.getStringExtra("apk_path") ?: return
        val apkFile = File(filePath)
        if (!apkFile.exists()) return
        try {
            val uri = FileProvider.getUriForFile(
                context,
                "${context.packageName}.fileprovider",
                apkFile
            )
            val installIntent = Intent(Intent.ACTION_VIEW).apply {
                setDataAndType(uri, "application/vnd.android.package-archive")
                addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
                addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
            }
            context.startActivity(installIntent)
        } catch (e: Exception) {
            android.util.Log.e("ApkInstallReceiver", "安装失败", e)
        }
    }
}
```

**新增通知栏构建辅助方法**：

```kotlin
/** 构建下载进度通知 */
private fun buildProgressNotification(progress: Float, version: String): android.app.Notification {
    return NotificationCompat.Builder(context, CHANNEL_ID)
        .setSmallIcon(android.R.drawable.stat_sys_download)
        .setContentTitle("正在下载快麦取货通 v${version}")
        .setContentText("${(progress * 100).toInt()}%")
        .setProgress(100, (progress * 100).toInt(), false)
        .setOngoing(true)
        .setSilent(true)
        .build()
}

/** 构建下载完成通知（点击安装） */
private fun buildCompleteNotification(apkFile: File, version: String): android.app.Notification {
    // 点击通知触发安装的 PendingIntent
    val installIntent = Intent(context, ApkInstallReceiver::class.java).apply {
        putExtra("apk_path", apkFile.absolutePath)
    }
    val pendingIntent = PendingIntent.getBroadcast(
        context,
        INSTALL_REQUEST_CODE,
        installIntent,
        PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
    )

    return NotificationCompat.Builder(context, CHANNEL_ID)
        .setSmallIcon(android.R.drawable.stat_sys_download_done)
        .setContentTitle("快麦取货通 v${version} 下载完成")
        .setContentText("点击安装更新")
        .setContentIntent(pendingIntent)
        .setAutoCancel(true)
        .setSilent(false)
        .build()
}

/** 构建失败通知 */
private fun buildFailedNotification(message: String): android.app.Notification {
    return NotificationCompat.Builder(context, CHANNEL_ID)
        .setSmallIcon(android.R.drawable.stat_sys_warning)
        .setContentTitle("下载失败")
        .setContentText(message.take(60))
        .setAutoCancel(true)
        .setSilent(false)
        .build()
}
```

**修改 `downloadApk` 方法**（在下载循环中增加通知更新）：

```kotlin
fun downloadApk(info: AppVersionResponse) {
    if (!_isDownloading.compareAndSet(false, true)) return
    _downloadState.value = DownloadState.Idle
    Thread {
        try {
            try {
                val dir = context.getExternalFilesDir(Environment.DIRECTORY_DOWNLOADS)
                    ?: File(context.cacheDir, "update")
                dir.mkdirs()
                val apkFile = File(dir, "快麦取货通-${info.latestVersion}.apk")
                if (apkFile.exists() && info.apkSize > 0 && apkFile.length() == info.apkSize) {
                    _downloadState.value = DownloadState.Completed(apkFile)
                    return@Thread
                }
                val request = Request.Builder().url(info.downloadUrl).build()
                val response = trustAllClient.newCall(request).execute()
                if (!response.isSuccessful) {
                    val msg = "HTTP ${response.code}"
                    _downloadState.value = DownloadState.Failed(msg)
                    showNotificationFailed(msg)
                    return@Thread
                }
                val body = response.body ?: run {
                    _downloadState.value = DownloadState.Failed("响应体为空")
                    showNotificationFailed("响应体为空")
                    return@Thread
                }
                val totalBytes = body.contentLength()
                var downloadedBytes = 0L
                val buffer = ByteArray(8192)
                FileOutputStream(apkFile).use { output ->
                    body.byteStream().use { input ->
                        var bytesRead: Int
                        while (input.read(buffer).also { bytesRead = it } != -1) {
                            output.write(buffer, 0, bytesRead)
                            downloadedBytes += bytesRead
                            if (totalBytes > 0) {
                                val progress = downloadedBytes.toFloat() / totalBytes
                                _downloadState.value = DownloadState.Downloading(progress, downloadedBytes, totalBytes)
                                // 更新通知栏进度（每50ms一次）
                                showNotificationProgress(progress, info.latestVersion)
                            }
                        }
                    }
                }
                if (info.apkSize > 0 && apkFile.length() != info.apkSize) {
                    apkFile.delete()
                    _downloadState.value = DownloadState.Failed("文件大小不匹配")
                    showNotificationFailed("文件大小不匹配")
                    return@Thread
                }
                saveToPublicDownloads(apkFile, info.latestVersion)
                _downloadState.value = DownloadState.Completed(apkFile)
                showNotificationCompleted(apkFile, info.latestVersion)
            } catch (e: IOException) {
                Log.e(TAG, "下载APK失败", e)
                val msg = e.message ?: "未知错误"
                _downloadState.value = DownloadState.Failed(msg)
                showNotificationFailed(msg)
            }
        } finally {
            _isDownloading.set(false)
        }
    }.start()
}
```

**新增通知展示辅助方法**：

```kotlin
private fun showNotificationProgress(progress: Float, version: String) {

private fun showNotificationCompleted(apkFile: File, version: String) {

        notificationManager.notify(NOTIFICATION_ID, buildFailedNotification(message))
```

### 修改 4：`MainActivity.kt` — 增加 Toast 反馈

**修改 collector 部分**（L122-132）：

```kotlin
lifecycleScope.launch {
    appUpdateManager.downloadState.collect { state ->
        when (state) {
            is DownloadState.Completed -> {
                // 尝试自动安装（兼容未点击通知的情况）
                appUpdateManager.installApk(state.file)
                showUpdateDialog = false
                updateInfo = null
                android.widget.Toast.makeText(
                    this@MainActivity,
                    "APK已保存到系统下载目录，可手动安装",
                    android.widget.Toast.LENGTH_LONG
                ).show()
            }
            is DownloadState.Failed -> {
                isDownloading = false
                android.widget.Toast.makeText(
                    this@MainActivity,
                    "下载失败: ${state.message}",
                    android.widget.Toast.LENGTH_LONG
                ).show()
            }
            else -> {} // Idle, Downloading 保持现状
        }
    }
}
```

## 四、用户行为流程

```
用户点击"立即更新"
  (Android 13+ 自动请求通知权限)
  → 通知栏出现: "正在下载快麦取货通 v2.5 - [========          ] 50%"
  → 弹窗显示进度条（同原来）

用户关闭弹窗（非强制更新）或回到桌面:
  → 通知栏继续显示下载进度  ← 新行为
  → 下载完成后台继续

下载完成:
  → 通知更新: "快麦取货通 v2.5 下载完成 - 点击安装更新"  ← 新行为
  → 系统通知震动/响铃
  → Toast: "APK已保存到系统下载目录，可手动安装"  ← 新行为
  → 自动尝试安装（兼容旧行为）
  → 用户点击通知 → 系统安装器打开
  → 用户打开文件管理器 → 在 Downloads 中找到 APK

下载失败:
  → 通知: "下载失败 - ..."  ← 新行为
  → Toast: "下载失败: ..."  ← 新行为
  → 按钮恢复可重试
```

## 五、回归风险

| # | 风险 | 说明 | 等级 |
|:-:|:-----|:------|:----:|
| 1 | 通知权限被拒绝（A13+） | `try-catch` 保护，不影响下载本身 | ✅ |
| 2 | `PendingIntent.FLAG_IMMUTABLE` 兼容性（A12-） | API 23+ 支持，minSdk=23 | ✅ |
| 3 | 通知渠道已存在时重复创建 | Android 自动覆盖 | ✅ |
| 4 | 通知频繁更新导致性能问题 | 每 `8192 bytes` 约 50ms 一次，与 StateFlow 同频 | ✅ |
| 5 | `BroadcastReceiver` 上下文 | Manifest 注册的应用级 | ✅ |
| 6 | `notificationManager.notify` 后台线程 | `mainThread` 只在 A14+ 强制，流式下载在子线程 | ⚠️ 可能需要 `Handler` 切主线程 |

**风险 6 处理**：在 Android 14+，`NotificationManager.notify()` 必须在主线程调用。解决方案：使用 `Handler(Looper.getMainLooper()).post { notificationManager.notify(...) }`。

```kotlin
private val mainHandler = android.os.Handler(android.os.Looper.getMainLooper())

private fun showNotificationProgress(progress: Float, version: String) {
    mainHandler.post {
        try {
            notificationManager.notify(NOTIFICATION_ID, buildProgressNotification(progress, version))
        } catch (_: Exception) {}
    }
}
```
